index로 매칭되기 때문에 data-js="accordion"안에 어떤형식이든 순서만 맞춰주면됨

Data type

data-multi=true //이전 패널이 닫히지 않음 (default는 닫힘)

SCRIPT

Use

$('[data-js="accordion"]').accordion();
$('[data-js="accordion"]').data('plugin_accordion').mothod();

Options

defaults = {
    event : 'click',    // 'mouseenter focusin'
    speed : 0.3,
    container : '[data-js="accordion"]',
    anchorEl : '[data-js="accordion__anchor"]',
    panelEl : '[data-js="accordion__panel"]',
    activeClassName : 'is-active',
    disabledClassName : 'is-disabled',
    multi : false,
    onLoadAccordion : null,
    onCompleteAccordion : null,
    onDestroyAccordion : null
}

Methods

dropdownOpen( index ),
dropdownClose( index)

Events

$('[data-js="accordion"]').on('ACCORDION_OPEN_START', function( e, data ){});
$('[data-js="accordion"]').on('ACCORDION_OPEN_END', function( e, data ){});
$('[data-js="accordion"]').on('ACCORDION_CLOSE_START', function( e, data ){});
$('[data-js="accordion"]').on('ACCORDION_CLOSE_END', function( e, data ){});

event target의 index, anchor, panel이 data로 전달

_END 이벤트는 onCompleteAccordion 옵션을 지정하지 않았을경우만 발생
